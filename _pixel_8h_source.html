292 template <class Rep, class Period>
293 std::future<std::shared_ptr<const Messages::PixelMessage>> sendAndWaitForResponseAsync(
294 Messages::MessageType type,
295 Messages::MessageType responseType,
296 std::chrono::duration<Rep, Period> timeout = std::chrono::seconds(5))
297 {
298 std::promise<std::shared_ptr<const Messages::PixelMessage>> responsePromise{};
299
300 MessageCallback callback{ [&responseType, &responsePromise](auto msg)
301 {
302 if (msg->type == responseType)
303 {
304 responsePromise.set_value(msg);
305 }
306 } };
307
308 const auto cbIndex = _internalMsgCbs.add(callback);
309
310 const auto sendResult = co_await sendMessageAsync(type);
311
312 std::shared_ptr<const Messages::PixelMessage> response{};
313 if (sendResult)
314 {
315 auto respFuture = responsePromise.get_future();
316 if (respFuture.wait_for(timeout) == std::future_status::ready)
317 {
318 response = respFuture.get();
319 }
320 }
321
322 _internalMsgCbs.remove(cbIndex);
323
324 co_return response;
325 }
